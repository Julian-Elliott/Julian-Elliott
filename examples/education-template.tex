% Current Education (if applicable)
\newcommand{\currentEducationEntry}{%
\cvevent{Current Degree/Program}{Institution Name}{Start Year -- Expected Graduation}{Location}
\begin{itemize}
  \item Relevant coursework: [Course 1], [Course 2], [Course 3]
  \item Projects: [Brief description of key projects]
  \item Expected graduation: [Month Year]
\end{itemize}
}

% Completed Degree
\newcommand{\degreeEntry}{%
\cvevent{Degree Name (e.g., BSc Computer Science)}{University Name}{Graduation Year}{Location}
\begin{itemize}
  \item Relevant coursework: [Relevant Courses]
  \item [Achievement/Honor if applicable]
  \item [Key project or thesis topic if relevant]
\end{itemize}
}

% Professional Development/Bootcamp
\newcommand{\bootcampEntry}{%
\cvevent{Program Name (e.g., Data Science Bootcamp)}{Institution/Provider}{Duration}{Format (Online/In-person)}
\begin{itemize}
  \item [Duration] intensive program in [subject area]
  \item Key skills: [Skill 1], [Skill 2], [Skill 3]
  \item Capstone project: [Brief description]
\end{itemize}
}
