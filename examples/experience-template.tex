% Example Experience Template
% Copy this structure and adapt it for your own experience

% Current Position Example
\newcommand{\currentPositionEntry}{%
\cvevent{Your Current Job Title}{Company Name}{Start Date -- Present}{Location}
\begin{itemize}
  \item Achievement with quantified result (\textbf{X\% improvement}).
  \item Led initiative that delivered \textbf{[quantified impact]}.
  \item Developed [specific solution] for [business problem].
  \item Mentored [number] team members in [specific skills].
  \item Key skills: [Skill 1], [Skill 2], [Skill 3]
\end{itemize}
}

% Previous Position Example
\newcommand{\previousPositionEntry}{%
\cvevent{Previous Job Title}{Company Name}{Start Date -- End Date}{Location}
\begin{itemize}
  \item Accomplished [specific task] resulting in [quantified outcome].
  \item Implemented [solution] that improved [metric] by [percentage].
  \item Collaborated with [stakeholders] to achieve [goal].
  \item Key skills: [Relevant Skills]
\end{itemize}
}
