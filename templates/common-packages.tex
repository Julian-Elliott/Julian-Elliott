% LaTeX Common Packages and Settings
% This file contains shared package imports and configurations
% Include this in all CV templates for consistency

% Essential packages for modern LaTeX
\usepackage[utf8]{inputenc}         % Input encoding
\usepackage[T1]{fontenc}            % Font encoding
% Skip microtype for XeLaTeX/LuaLaTeX due to compatibility issues
\iftutex\else
  \usepackage{microtype}            % Improved typography (PDFLaTeX only)
\fi
\usepackage{babel}                  % Language support

% Color and graphics
\usepackage{xcolor}                 % Advanced color support
\usepackage{graphicx}               % Graphics inclusion
\usepackage{tikz}                   % Advanced graphics

% Layout and spacing
\usepackage{geometry}               % Page layout
\usepackage{setspace}               % Line spacing
\usepackage{enumitem}               % List customization
\usepackage{parskip}                % Paragraph spacing
\usepackage{paracol}                % For multi-column layouts
\usepackage{tcolorbox}              % For colored and framed boxes (skill tiles)
\tcbuselibrary{skins}               % For enhanced tcolorbox styling

% Links and hyperref (load last)
\usepackage{url}                    % URL formatting
\usepackage{hyperref}               % Hyperlinks

% Configure hyperref for better PDF output
\hypersetup{
    colorlinks=true,
    linkcolor=blue,
    urlcolor=blue,
    citecolor=blue,
    filecolor=blue,
    pdfpagemode=UseNone,
    pdfstartview=FitH,
    pdftitle={CV - \myname},
    pdfauthor={\myname},
    pdfsubject={Curriculum Vitae},
    pdfkeywords={CV, Resume, \myname},
    pdfcreator={LaTeX with CV Template System},
        pdfproducer={XeLaTeX}
}

% Improved list formatting
\setlist{
    itemsep=0pt,
    parsep=0pt,
    topsep=0pt,
    partopsep=0pt,
    leftmargin=*
}

% Better paragraph formatting
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.5\baselineskip}

% Skill Tile Command - Minimal design for optimal space usage
\newcommand{\skilltile}[1]{%
  \tcbox[%
    colback=LightGrey!8,  % Very subtle background
    colframe=VividPurple, % Accent color border
    coltext=SlateGrey,    % Text color
    fontupper=\scriptsize, % Small but readable font
    halign=center,        % Center align text
    valign=center,        % Center align text vertically
    boxsep=0pt,           % No internal padding
    left=1.5pt, right=1.5pt, % Minimal horizontal padding
    top=0.5pt, bottom=0.5pt, % Minimal vertical padding
    arc=1pt,              % Small rounded corners
    boxrule=0.2pt,        % Very thin border
    nobeforeafter,        % No extra vertical space
    on line,              % Inline placement
    tcbox width=auto      % Auto width to content
  ]{#1}\hspace{1.5pt}}    % Small space after tile
